36  template<typename MetricSet, typename MetricProxy, typename Point>
37  size_t populate_metric_average_by_cycle(const MetricSet& metrics,
38  MetricProxy& proxy,
39  const model::plot::filter_options& options,
40  const constants::metric_type type,
41  model::plot::data_point_collection<Point>& points)
42  {
43  const size_t max_cycle = metrics.max_cycle();
44  points.assign(max_cycle, Point());
45  const float dummy_x = 1;
46  for(typename MetricSet::const_iterator b = metrics.begin(), e = metrics.end();b != e;++b)
47  {
48  if(!options.valid_tile(*b)) continue;
49  const float val = proxy(*b, type);
50  if(std::isnan(val)) continue;
51  points[b->cycle()-1].add(dummy_x, val);
52  }
53  size_t index = 0;
54  for(size_t cycle=0;cycle<max_cycle;++cycle)
55  {
56  if(static_cast<size_t>(points[cycle].x()) == 0) continue;
57  const float avg = points[cycle].y()/points[cycle].x();
58  points[index].set(static_cast<float>(cycle+1), avg);
59  ++index;
60  }
61  points.resize(index);
62  return max_cycle;
63  }
73  template<typename MetricSet, typename MetricProxy>
74  size_t populate_candle_stick_by_cycle(const MetricSet& metrics,
75  MetricProxy& proxy,
76  const model::plot::filter_options& options,
77  const constants::metric_type type,
78  model::plot::data_point_collection<model::plot::candle_stick_point>& points)
79  {
80  const size_t max_cycle = metrics.max_cycle();
81  const size_t tile_count = static_cast<size_t>(std::ceil(static_cast<float>(metrics.size())/max_cycle));
82  std::vector< std::vector<float> > tile_by_cycle(max_cycle);
83  for(size_t i=0;i<max_cycle;++i) tile_by_cycle[i].reserve(tile_count);
84  std::vector<float> outliers;
85  outliers.reserve(10); // TODO: use as flag for keeping outliers
86 
87  for(typename MetricSet::const_iterator b = metrics.begin(), e = metrics.end();b != e;++b)
88  {
89  if(!options.valid_tile(*b)) continue;
90  const float val = proxy(*b, type);
91  if(std::isnan(val)) continue;
92  tile_by_cycle[b->cycle()-1].push_back(val);
93  }
94  points.resize(max_cycle);
95  size_t j=0;
96  for(size_t cycle=0;cycle<max_cycle;++cycle)
97  {
98  if(tile_by_cycle[cycle].empty())continue;
99  plot_candle_stick(points[j],
100  tile_by_cycle[cycle].begin(),
101  tile_by_cycle[cycle].end(),
102  static_cast<float>(cycle+1),
103  outliers);
104  ++j;
105  }
106  points.resize(j);
107  return max_cycle;
108  }

157  const model::plot::filter_options& options,

159  throw(model::index_out_of_bounds_exception,
160  model::invalid_metric_type,
161  model::invalid_channel_exception,
162  model::invalid_filter_option,
163  model::invalid_read_exception)
164  {
165  data.clear();
166  if(!options.all_cycles())
167  INTEROP_THROW(model::invalid_filter_option, "Filtering by cycle is not supported");
168  if(!options.all_reads())
169  INTEROP_THROW(model::invalid_filter_option, "Filtering by read is not supported");
170  if(!utils::is_cycle_metric(type))
171  INTEROP_THROW(model::invalid_filter_option, "Only cycle metrics are supported");
172  options.validate(type, metrics.run_info());
173  size_t max_cycle=0;
174  switch(logic::utils::to_group(type))
175  {
176  case constants::Extraction:
177  {
178  if(options.all_channels(type))
179  {
180  setup_series_by_channel(metrics.run_info().channels(), data);
181  for(size_t i=0;i<data.size();++i)
182  {
183  metric::metric_value<model::metrics::extraction_metric> proxy(i);
184  max_cycle = populate_metric_average_by_cycle(
185  metrics.get_set<model::metrics::extraction_metric>(),
186  proxy,
187  options,
188  type,
189  data[i]
190  );
191  }
192  }
193  else
194  {
195  data.assign(1, model::plot::series<Point>());
196  const size_t channel = options.channel();
197  metric::metric_value<model::metrics::extraction_metric> proxy(channel);
198  max_cycle = populate_candle_stick_by_cycle(
199  metrics.get_set<model::metrics::extraction_metric>(),
200  proxy,
201  options,
202  type,
203  data[0]);
204  }
205  break;
206  }
207  case constants::CorrectedInt:
208  {
209  if(options.all_bases(type))
210  {
211  setup_series_by_base(data);
212  for(size_t i=0;i<data.size();++i)
213  {
214  metric::metric_value<model::metrics::corrected_intensity_metric> proxy(
215  static_cast<constants::dna_bases>(i));
216  max_cycle = populate_metric_average_by_cycle(
217  metrics.get_set<model::metrics::corrected_intensity_metric>(),
218  proxy,
219  options,
220  type,
221  data[i]
222  );
223  }
224  }
225  else
226  {
227  data.assign(1, model::plot::series<Point>());
228  const constants::dna_bases base = options.dna_base();
229  metric::metric_value<model::metrics::corrected_intensity_metric> proxy(base);
230  max_cycle = populate_candle_stick_by_cycle(
231  metrics.get_set<model::metrics::corrected_intensity_metric>(),
232  proxy,
233  options,
234  type,
235  data[0]);
236  }
237  break;
238  }
239  case constants::Q:
240  {
241  data.assign(1, model::plot::series<Point>());
242 
243  typedef model::metrics::q_collapsed_metric metric_t;
244  metric::metric_value<metric_t> proxy2;
245  if(0 == metrics.get_set<metric_t>().size())
246  logic::metric::create_collapse_q_metrics(metrics.get_set<model::metrics::q_metric>(),
247  metrics.get_set<metric_t>());
248  max_cycle = populate_candle_stick_by_cycle(
249  metrics.get_set<metric_t>(),
250  proxy2,
251  options,
252  type,
253  data[0]);
254  break;
255  }
256  case constants::Error://TODO: skip last cycle of read for error metric
257  {
258  data.assign(1, model::plot::series<Point>());
259  metric::metric_value<model::metrics::error_metric> proxy3;
260  max_cycle = populate_candle_stick_by_cycle(
261  metrics.get_set<model::metrics::error_metric>(),
262  proxy3,
263  options,
264  type,
265  data[0]);
266  break;
267  }
268  default:
269  INTEROP_THROW(model::invalid_metric_type, "Invalid metric group");
270  }
271  if(type != constants::FWHM)
272  {
273  auto_scale_y(data);
274  if(type == constants::ErrorRate)
275  {
276  data.set_yrange(0, std::min(5.0f, data.y_axis().max()));
277  }
278  else if(type == constants::BasePercent)
279  {
280  data.set_yrange(0, std::max(50.0f, data.y_axis().max()));
281  }
282  }
283  else data.set_yrange(0, 6);
284  data.set_xrange(0, static_cast<float>(max_cycle+2));
285  data.set_xlabel("Cycle");
286  data.set_ylabel(utils::to_description(type));
287 
288  std::string title = metrics.run_info().flowcell().barcode();
289  if(title != "") title += " ";
290  title += options.lane_description();
291  if(logic::utils::is_channel_metric(type))
292  title += " " + options.channel_description(metrics.run_info().channels());
293  if(logic::utils::is_base_metric(type))
294  title += " " + options.base_description();
295  if(metrics.run_info().flowcell().surface_count()>1 && options.is_specific_surface())
296  title += " " + options.surface_description();
297  data.set_title(title);
298  }
