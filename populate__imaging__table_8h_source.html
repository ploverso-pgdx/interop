65  inline const std::vector<std::string>& imaging_table_column_names()
66  {
67  /* For every entry in INTEROP_IMAGING_COLUMN_TYPES
68  *
69  * Convert each entry to a string name
70  *
71  * Example:
72  * INTEROP_TUPLE6(Lane, metric_base::base_metric, lane, Void, UInt, IdType) ->
73  * "Lane"
74  *
75  * @see INTEROP_IMAGING_COLUMN_TYPES
76  */
77 # define INTEROP_TUPLE7(Id, Ignored1, Ignored2, Ignored3, Ignored4, Ignored5, Ignored6) #Id,
78  // The list of strings below ends with a ',', e.g. "Lane", "Tile", "Cycle",
79  // This is not allowed in C++, so I added a dummy string to the end.
80  static const std::string name_types[] = {INTEROP_IMAGING_COLUMN_TYPES "Dummy"};// Added dummy as a work around
81 # undef INTEROP_TUPLE7
82 
83  static const std::vector<std::string> tmp(name_types, name_types+util::length_of(name_types)-1);// remove dummy
84  return tmp;
85  }

164  inline void populate_column_headers(std::vector<model::table::column_header>& headers,
165  const std::vector<std::string>& channels,
166  const std::vector<bool>& filled)
167  throw(model::invalid_column_type)
168  {
169  const std::vector<std::string>& names(imaging_table_column_names());
170  headers.clear();
171  headers.reserve(names.size()+channels.size()+constants::NUM_OF_BASES);
172  if(filled.size() != names.size())
173  INTEROP_THROW(model::invalid_column_type, "Filled vector does not match number of column names");
174  for(size_t i=0;i<names.size();++i)
175  {
176  if(!filled[i]) continue;
177  const std::string title = to_header(names[i]);
178  switch(to_data_type(i))
179  {
180  case constants::IdType:
181  case constants::StructType:
182  case constants::ValueType:
183  {
184  headers.push_back(title);
185  break;
186  }
187  case constants::BaseArray:
188  {
189  for(size_t base = 0;base < constants::NUM_OF_BASES;++base)
190  headers.push_back(
191  model::table::column_header(title,
192  constants::to_string(static_cast<constants::dna_bases>(base))));
193 
194  break;
195  }
196  case constants::ChannelArray:
197  {
198  for(size_t channel = 0;channel < channels.size();++channel)
199  headers.push_back(model::table::column_header(title, channels[channel]));
200  break;
201  }
202  default:
203  {
204  INTEROP_THROW( model::invalid_column_type, "Column index does not have a type");
205  }
206  }
207  }
208  }
