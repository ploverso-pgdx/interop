52  inline std::string scale_histogram(std::vector<float>& histogram)
53  {
54  float max_height = 0;

86  template<typename Point>
87  float plot_unbinned_histogram(const std::vector<float>& histogram,
88  model::plot::data_point_collection<Point>& points)
89  {
90  points.resize(histogram.size());
91  size_t point_index = 0;
92  for(size_t i=0;i<histogram.size();++i)
93  {
94  if(histogram[i] == 0) continue;
95  points[point_index].set(static_cast<float>(i+1), histogram[i]);
96  ++point_index;
97  }
98  points.resize(point_index);
99  if(point_index == 0) return 0;
100  return static_cast<float>(points[point_index-1].x()+1);
101  }
110  template<typename I, typename Point>
111  float plot_binned_histogram(I beg,
112  I end,
113  const std::vector<float>& histogram,
114  model::plot::data_point_collection<Point>& points)
115  {
116  float max_x_value = 0;
117  points.resize(std::distance(beg, end));
118  size_t point_index = 0;
119  if(points.size() == histogram.size()) //Compressed
120  {
121  for (size_t i=0; beg != end; ++beg, ++i)
122  {
123  INTEROP_ASSERT(i < histogram.size());
124  if(histogram[i] == 0)continue;
125  points[point_index].set(beg->lower(), histogram[i], static_cast<float>(beg->upper()-beg->lower()+1));
126  max_x_value = std::max(max_x_value, points[point_index].x()+points[point_index].width());
127  ++point_index;
128  }
129  }
130  else // Uncompressed
131  {
132  for (size_t i=0; beg != end; ++beg, ++i)
133  {
134  const size_t bin = static_cast<size_t>(beg->value())-1;
135  INTEROP_ASSERTMSG(bin < histogram.size(), bin << " < " << histogram.size());
136  if(histogram[bin] == 0)continue;
137  points[point_index].set(beg->lower(), histogram[bin], static_cast<float>(beg->upper()-beg->lower()+1));
138  max_x_value = std::max(max_x_value, points[point_index].x()+points[point_index].width());
139  ++point_index;
140  }
141  }
142  points.resize(point_index);
143  return max_x_value;
144  }
