32  template<typename MetricSet, typename MetricProxy, typename Point>
33  void populate_candle_stick_by_lane(const MetricSet& metrics,
34  MetricProxy& proxy,
35  const model::plot::filter_options& options,
36  const constants::metric_type type,
37  model::plot::data_point_collection<Point>& points)
38  {
39  const size_t lane_count = metrics.max_lane();
40  const size_t tile_count = static_cast<size_t>(std::ceil(static_cast<float>(metrics.size())/lane_count));
41  std::vector< std::vector<float> > tile_by_lane(metrics.max_lane());
42  for(size_t i=0;i<tile_by_lane.size();++i) tile_by_lane[i].reserve(tile_count); // optimize using lane ids
43  std::vector<float> outliers;
44  outliers.reserve(10);
45 
46  for(typename MetricSet::const_iterator b = metrics.begin(), e = metrics.end();b != e;++b)
47  {
48  if(!options.valid_tile(*b)) continue;
49  const float val = proxy(*b, type);
50  if(std::isnan(val)) continue;
51  tile_by_lane[b->lane()-1].push_back(val);
52  }
53  points.resize(lane_count);
54  size_t offset=0;
55  for(size_t i=0;i<tile_by_lane.size();++i)
56  {
57  if(tile_by_lane[i].empty()) continue;
58  const float lane = static_cast<float>(i+1);
59  plot_candle_stick(points[offset], tile_by_lane[i].begin(), tile_by_lane[i].end(), lane, outliers);
60  ++offset;
61  }
62  points.resize(offset);
63  }
