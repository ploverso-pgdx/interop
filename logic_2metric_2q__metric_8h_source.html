65  void populate_cumulative_distribution_t(model::metric_base::metric_set<QMetric>& metric_set)
66  throw( model::index_out_of_bounds_exception )
67  {
68  if(metric_set.size()==0) return;
69  typedef QMetric q_metric;
70  typedef model::metric_base::metric_set<q_metric> q_metric_set;
71  typedef typename q_metric_set::id_vector id_vector;
72  typedef typename q_metric_set::uint_t uint_t;
73  typedef typename id_vector::const_iterator const_id_iterator;
74  id_vector lane_ids = metric_set.lanes();
75  for(const_id_iterator lane_beg = lane_ids.begin(), lane_end = lane_ids.end();lane_beg != lane_end;++lane_beg)
76  {
77  const uint_t lane_id = *lane_beg;
78  id_vector tile_ids = metric_set.tile_numbers_for_lane(lane_id);
79  for(const_id_iterator tile_beg = tile_ids.begin(), tile_end = tile_ids.end();tile_beg != tile_end;++tile_beg)
80  {
81  const uint_t tile_id = *tile_beg;
82  size_t prev_idx = metric_set.find(lane_id, tile_id, 1);
83  if(prev_idx >= metric_set.size()) continue;
84  QMetric& metric = metric_set.at(prev_idx);
85  metric.accumulate(metric);
86  const uint_t second_cycle_start = 2; // We have to accumulate the first cycle with itself, and every
87  // subsequent with the previous cycle.
88  // Also this is not 0-indexed, so we start with 2, the 2nd cycle
89  for(uint_t cycle = second_cycle_start;cycle <= metric_set.max_cycle();++cycle)
90  {
91  const size_t cur_idx = metric_set.find(lane_id, tile_id, cycle);
92  if(cur_idx>=metric_set.size() || prev_idx>=metric_set.size())
93  continue;// TODO: if this happens zero out following q-scores
94  metric_set.at(cur_idx).accumulate(metric_set.at(prev_idx));
95  prev_idx=cur_idx;
96  }
97  }
98  }
99  }

161  inline size_t count_legacy_q_score_bins(const model::metric_base::metric_set<model::metrics::q_metric>& q_metric_set)
162  {
163  // 0 is a sentinel that indicates legacy binning is not required
164  if(q_metric_set.version() > 4) return 0; // Version 5 and later do not require legacy binning
165  if(!q_metric_set.bins().empty()) return 0; // If the metrics already have a header they do not require binning
166 
167  const size_t max_bin_count = 7;
168  model::metric_base::metric_set<model::metrics::q_metric>::const_iterator beg = q_metric_set.begin(),
169  end=q_metric_set.end();
170  if(beg == end) return 0 ;
171  typedef model::metrics::q_metric::uint_t uint_t;
172  std::set<uint_t> bins_found;
173  for(;beg != end;++beg)
174  {
175  for(uint_t i=0;i<static_cast<uint_t>(beg->qscoreHist().size());++i)
176  if(beg->qscoreHist()[i] > 0) bins_found.insert(i);
177  if(bins_found.size() > max_bin_count) break; // Number of bins greater than 7 indicates this is unbinned
178  }
179  return bins_found.size();
180  }
