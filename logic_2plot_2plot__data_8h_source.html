22 void auto_scale(model::plot::plot_data<Point>& data, const bool zero_min=true, const float scale_y_max=1.1f)
23 {
24  typedef typename model::plot::plot_data<Point>::const_iterator const_series_iterator;
25  typedef typename model::plot::series<Point>::const_iterator const_point_iterator;
26  float ymin = std::numeric_limits<float>::max();
27  float xmin = std::numeric_limits<float>::max();
28  float ymax = -std::numeric_limits<float>::max();
29  float xmax = -std::numeric_limits<float>::max();
30  for(const_series_iterator cur_series = data.begin(), series_end = data.end();cur_series != series_end;++cur_series)
31  {
32  for(const_point_iterator cur_pt = cur_series->begin(), end_pt = cur_series->end();cur_pt != end_pt;++cur_pt)
33  {
34  ymax = std::max(ymax, cur_pt->max_value());
35  xmax = std::max(xmax, cur_pt->x());
36  if(zero_min) continue;
37  ymin = std::min(ymin, cur_pt->min_value());
38  xmin = std::min(xmin, cur_pt->x());
39  }
40  }
41  if(ymin == std::numeric_limits<float>::max()) ymin = 0;
42  if(xmin == std::numeric_limits<float>::max()) xmin = 0;
43  ymax = (ymax == -std::numeric_limits<float>::max()) ? 0 : scale_y_max * ymax + 0.0001f;
44  if(xmax == -std::numeric_limits<float>::max()) xmax = 0;
45  data.set_range(xmin, xmax, ymin, ymax);
46 }
