357  template<typename F, typename I>
358  F percentile_sorted(I beg, I end, const size_t percentile)
359  {
360  INTEROP_ASSERT(beg != end);
361  INTEROP_ASSERT(percentile > 0 && percentile <= 100);
362  const size_t n = static_cast<size_t>(std::distance(beg, end));
363  if (n == 0) return std::numeric_limits<F>::quiet_NaN();
364  size_t nth_index = percentile * n / 100;
365  if ((n * percentile / 100.0f - nth_index) < 0.5f)
366  {
367  if (nth_index == 0) return *beg;
368  nth_index--;
369  }
370  if (nth_index >= (n - 1)) return *(end - 1);
371  const F y1 = *(beg + nth_index);
372  const F y2 = *(beg + nth_index + 1);
373  const F x1 = 100.0f * (nth_index + 0.5f) / n;
374  const F x2 = 100.0f * (nth_index + 0.5f + 1) / n;
375  return interpolate_linear(y1, y2, x1, x2, static_cast<float>(percentile));
376  }
