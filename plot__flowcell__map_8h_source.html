43  if(beg == end) return;

54  layout.tile_count(),

57  beg->tile(),
58  val);
59  values_for_scaling.push_back(val);
60  }

84 
85  options.validate(type, metrics.run_info());
86 
87  if(utils::is_cycle_metric(type) && options.all_cycles())
88  INTEROP_THROW( model::invalid_filter_option, "All cycles is unsupported");
89  if(utils::is_read_metric(type) && options.all_reads() && metrics.run_info().reads().size() > 1)
90  INTEROP_THROW( model::invalid_filter_option, "All reads is unsupported");
91  switch(logic::utils::to_group(type))
92  {
93  case constants::Tile:
94  {
95  typedef model::metrics::tile_metric metric_t;
96  typedef model::metric_base::metric_set<metric_t> metric_set_t;
97  const metric_set_t& metric_set = metrics.get_set<metric_t>();
98  metric::metric_value<metric_t> proxy(options.read());
99  populate_flowcell_map(metric_set.begin(), metric_set.end(), proxy, type, layout, options, data,
100  values_for_scaling);
101  break;
102  }
103  case constants::Extraction:
104  {
105  typedef model::metrics::extraction_metric metric_t;
106  typedef model::metric_base::metric_set<metric_t> metric_set_t;
107  const metric_set_t& metric_set = metrics.get_set<metric_t>();
108  const size_t channel = options.channel();
109  if(options.all_channels(type))
110  INTEROP_THROW(model::invalid_filter_option, "All channels is unsupported");
111  metric::metric_value<metric_t> proxy(channel);
112  populate_flowcell_map(metric_set.begin(), metric_set.end(), proxy, type, layout, options, data,
113  values_for_scaling);
114  break;
115  }
116  case constants::CorrectedInt:
117  {
118  typedef model::metrics::corrected_intensity_metric metric_t;
119  typedef model::metric_base::metric_set<metric_t> metric_set_t;
120  const metric_set_t& metric_set = metrics.get_set<metric_t>();
121  const constants::dna_bases base = options.dna_base();
122  if(options.all_bases(type))
123  INTEROP_THROW( model::invalid_filter_option, "All bases is unsupported");
124  metric::metric_value<metric_t> proxy(base);
125  populate_flowcell_map(metric_set.begin(), metric_set.end(), proxy, type, layout, options, data,
126  values_for_scaling);
127  break;
128  }
129  case constants::Q:
130  {
131  typedef model::metrics::q_collapsed_metric metric_t;
132  typedef model::metric_base::metric_set<metric_t> metric_set_t;
133  metric_set_t &metric_set = metrics.get_set<metric_t>();
134  if(0 == metric_set.size())
135  {
136  logic::metric::create_collapse_q_metrics(metrics.get_set<model::metrics::q_metric>(), metric_set);
137  if(type == constants::AccumPercentQ20 || type == constants::AccumPercentQ30)
138  logic::metric::populate_cumulative_distribution(metric_set);
139  }
140  metric::metric_value<metric_t> proxy;
141  populate_flowcell_map(metric_set.begin(), metric_set.end(), proxy, type, layout, options, data,
142  values_for_scaling);
143  break;
144  }
145  case constants::Error:
146  {
147  typedef model::metrics::error_metric metric_t;
148  typedef model::metric_base::metric_set<metric_t> metric_set_t;
149  const metric_set_t& metric_set = metrics.get_set<metric_t>();
150  metric::metric_value<metric_t> proxy;
151  populate_flowcell_map(metric_set.begin(), metric_set.end(), proxy, type, layout, options, data,
152  values_for_scaling);
153  break;
154  }
155  default:
156  INTEROP_THROW( model::invalid_metric_type, "Unsupported metric type: " << constants::to_string(type));
157  };

159  if(!values_for_scaling.empty())

161  std::sort(values_for_scaling.begin(), values_for_scaling.end());
162  // TODO: Put this back

168  const float lower = values_for_scaling[size_t(0.25*values_for_scaling.size())];
169  const float upper = values_for_scaling[size_t(0.75*values_for_scaling.size())];
170  data.set_range(std::max(lower - 2 * (upper - lower), values_for_scaling[0]),
171  std::min(values_for_scaling.back(), upper + 2 * (upper - lower)));
172  }
173  else data.set_range(0,0);
174  if(type == constants::ErrorRate) data.set_range(0, std::min(5.0f, data.saxis().max()));
175 
176  std::string title = metrics.run_info().flowcell().barcode();

178  title += utils::to_description(type);
179  data.set_title(title);
180

183  subtitle += options.surface_description() + " ";

186  subtitle += " " + options.channel_description(metrics.run_info().channels());
187  if(logic::utils::is_base_metric(type))
188  subtitle += " " + options.base_description();
189  if(logic::utils::is_read_metric(type))
190  subtitle += " " + options.read_description();
191  data.set_subtitle(subtitle);
192  data.set_label(utils::to_description(type));
