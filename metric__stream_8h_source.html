125  template<class MetricSet>
126  void read_metrics(std::istream &in, MetricSet &metrics)
127  {
128  typedef typename MetricSet::metric_type metric_t;
129  typedef typename metric_format_adapter<metric_t>::metric_t format_metric_t;
130  typedef metric_format_factory<format_metric_t> factory_t;
131  typedef typename factory_t::metric_format_map metric_format_map;
132  typedef typename metric_t::id_t id_t;
133  typedef std::map<id_t, size_t> offset_map_t;
134  metric_format_map &format_map = factory_t::metric_formats();
135 
136  if (!in.good()) INTEROP_THROW(incomplete_file_exception, "Empty file found");
137  const int version = in.get();
138  if (version == -1) INTEROP_THROW(incomplete_file_exception, "Empty file found");
139  if (format_map.find(version) == format_map.end())
140  INTEROP_THROW(bad_format_exception, "No format found to parse " <<
141  interop_basename<MetricSet>() << " with version: " << version << " of " << format_map.size() );
142  INTEROP_ASSERT(format_map[version]);
143  metrics.set_version(static_cast< ::int16_t>(version));
144  const std::streamsize record_size = format_map[version]->read_header(in, metrics);
145 
146  offset_map_t metric_offset_map;
147  while (in)
148  {
149  metric_t metric;// Moving this out breaks index metrics - TODO: fix by adding clear to index metrics (or move) or swap
150  const std::streamsize icount = format_map[version]->read_metric_id(in, metric);
151  if (in.fail())
152  {
153  if (icount == 0 && !metric_offset_map.empty()) break;
154  INTEROP_THROW(incomplete_file_exception, "Insufficient data read from the file, got: " << icount
155  << " != expected: " << record_size);
156  }
157 
158  if (metric_offset_map.find(metric.id()) == metric_offset_map.end())
159  {
160  const std::streamsize count = format_map[version]->read_metric(in, metric, metrics, true) + icount;
161  if (in.fail())
162  {
163  if (count == 0 && !metric_offset_map.empty()) break;
164  INTEROP_THROW(incomplete_file_exception, "Insufficient data read from the file, got: " << count
165  << " != expected: " << record_size);
166  }
167  if (count != record_size)
168  {
169  INTEROP_THROW(bad_format_exception, "Record does not match expected size!");
170  }
171  if (metric.lane() > 0 && (metric_t::CHECK_TILE_ID == 0 || metric.tile() > 0))
172  {
173  metric_offset_map[metric.id()] = metrics.size();
174  metrics.insert(metric.id(), metric);
175  metrics.metric_updated_at(metric_offset_map[metric.id()]);
176  }
177  }
178  else
179  {
180  const size_t offset = metric_offset_map[metric.id()];
181  INTEROP_ASSERTMSG(metrics.at(offset).lane() != 0, offset);
182  const std::streamsize count =
183  format_map[version]->read_metric(in, metrics.at(offset), metrics, false) + icount;
184  if (in.fail())
185  {
186  if (count == 0 && !metric_offset_map.empty()) break;
187  INTEROP_THROW(incomplete_file_exception, "Insufficient data read from the file, got: " << count
188  << " != expected: " << record_size);
189  }
190  if (count != record_size)
191  {
192  INTEROP_THROW(bad_format_exception, "Record does not match expected size!");
193  }
194  metrics.metric_updated_at(offset);
195  }
196  }
197  }
