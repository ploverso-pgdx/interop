83  void validate(const constants::metric_type type, const run::info& run_info, const bool check_ignored=false)const throw(model::invalid_filter_option)
84  {
85  if(m_naming_method == constants::UnknownTileNamingMethod)
86  INTEROP_THROW(model::invalid_filter_option, "Invalid tile naming method: Unknown");
87  if(m_naming_method > constants::TileNamingMethodCount)
88  INTEROP_THROW(model::invalid_filter_option, "Invalid tile naming method: exceeds total number");
89  if(m_naming_method != run_info.flowcell().naming_method())
90  INTEROP_THROW(model::invalid_filter_option, "Invalid tile naming method: does not match RunInfo.xml");
91 
92  if(!all_lanes() && m_lane > run_info.flowcell().lane_count())
93  INTEROP_THROW(model::invalid_filter_option, "Lane number exceeds total number of lanes" << m_lane << " > " << run_info.flowcell().lane_count());
94  if(is_specific_surface() && m_surface > run_info.flowcell().surface_count())
95  INTEROP_THROW(model::invalid_filter_option, "Surface number exceeds total number of surfaces" << m_surface << " > " << run_info.flowcell().surface_count());
96  if(!all_tile_numbers() && m_tile_number > run_info.flowcell().tile_count())
97  INTEROP_THROW(model::invalid_filter_option, "Tile number exceeds total number of tile numbers" << m_tile_number << " > " << run_info.flowcell().tile_count());
98  if(!all_swaths() && m_swath > run_info.flowcell().swath_count())
99  INTEROP_THROW(model::invalid_filter_option, "Swath number exceeds total number of swaths" << m_swath << " > " << run_info.flowcell().swath_count());
100  if(!all_sections() && m_section > run_info.flowcell().sections_per_lane())
101  INTEROP_THROW(model::invalid_filter_option, "Section number exceeds total number of sections" << m_section << " > " << run_info.flowcell().sections_per_lane());
102  if(logic::utils::is_base_metric(type))
103  {
104  if(!all_bases() && (m_base >= static_cast<dna_base_t>(constants::NUM_OF_BASES) || m_base < 0))
105  INTEROP_THROW(model::invalid_filter_option, "Base to filter is invalid: " << to_string(m_base));
106  }
107  if(logic::utils::is_cycle_metric(type))
108  {
109  if(!all_cycles() && m_cycle > run_info.total_cycles())
110  INTEROP_THROW(model::invalid_filter_option, "Cycle number exceeds total number of cycles" << m_cycle << " > " << run_info.total_cycles());
111  }
112  if(logic::utils::is_read_metric(type))
113  {
114  if(!all_reads() && m_read > run_info.reads().size())
115  INTEROP_THROW(model::invalid_filter_option, "Read number exceeds total number of reads" << m_read << " > " << run_info.reads().size());
116  }
117  if(logic::utils::is_channel_metric(type))
118  {
119  if(!all_channels() && static_cast<size_t>(m_channel) >= run_info.channels().size())
120  INTEROP_THROW(model::invalid_filter_option, "Channel number exceeds total number of channels" << m_channel << " > " << run_info.channels().size());
121  }
122  if(!check_ignored) return;
123  if(!logic::utils::is_base_metric(type) && !all_bases())
124  INTEROP_THROW(model::invalid_filter_option, "Invalid filter option base for metric " << constants::to_string(type));
125  if(!logic::utils::is_cycle_metric(type) && !all_cycles())
126  INTEROP_THROW(model::invalid_filter_option, "Invalid filter option cycle for metric " << constants::to_string(type));
127  if(!logic::utils::is_read_metric(type) && !all_reads())
128  INTEROP_THROW(model::invalid_filter_option, "Invalid filter option read for metric " << constants::to_string(type));
129  if(!logic::utils::is_channel_metric(type) && !all_channels())
130  INTEROP_THROW(model::invalid_filter_option, "Invalid filter option channel for metric " << constants::to_string(type));
131  //all_cycles
132  }
