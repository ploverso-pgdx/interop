200  void finalize_after_load(size_t count = std::numeric_limits<size_t>::max()) throw(io::format_exception,
201  model::invalid_tile_naming_method,
202  model::index_out_of_bounds_exception)
203  {
204  if (m_run_info.flowcell().naming_method() == constants::UnknownTileNamingMethod)
205  {
206  m_run_info.set_naming_method(
207  logic::metric::tile_naming_method_from_metric(get_set<metrics::tile_metric>()));
208  if (m_run_info.flowcell().naming_method() == constants::UnknownTileNamingMethod)
209  m_run_info.set_naming_method(
210  logic::metric::tile_naming_method_from_metric(get_set<metrics::extraction_metric>()));
211  if (m_run_info.flowcell().naming_method() == constants::UnknownTileNamingMethod)
212  m_run_info.set_naming_method(
213  logic::metric::tile_naming_method_from_metric(get_set<metrics::q_metric>()));
214  }
215  if (count == std::numeric_limits<size_t>::max())
216  {
217  if(get_set<q_metric>().size() > 0)
218  count = logic::metric::count_legacy_q_score_bins(get_set<q_metric>());
219  else if(get_set<q_by_lane_metric>().size())
220  count = logic::metric::count_legacy_q_score_bins(get_set<q_by_lane_metric>());
221  }
222  logic::metric::populate_legacy_q_score_bins(get_set<q_metric>().bins(), m_run_parameters.instrument_type(),
223  count);
224  if (get_set<q_metric>().size() > 0 && get_set<q_collapsed_metric>().size() == 0)
225  logic::metric::create_collapse_q_metrics(get_set<q_metric>(), get_set<q_collapsed_metric>());
226  if (get_set<q_metric>().size() > 0 && get_set<q_by_lane_metric>().size() == 0)
227  logic::metric::create_q_metrics_by_lane(get_set<q_metric>(), get_set<q_by_lane_metric>());
228  logic::metric::populate_cumulative_distribution(get_set<q_metric>());
229  logic::metric::populate_cumulative_distribution(get_set<q_by_lane_metric>());
230  logic::metric::populate_cumulative_distribution(get_set<q_collapsed_metric>());
231  INTEROP_ASSERT(get_set<q_metric>().size() == 0 || get_set<q_metric>().size() == get_set<q_collapsed_metric>().size());
232  if (m_run_info.channels().empty())
233  {
234  legacy_channel_update(m_run_parameters.instrument_type());
235  if (m_run_info.channels().empty())
236  INTEROP_THROW(io::format_exception,
237  "Channel names are missing from the RunInfo.xml, and RunParameters.xml does not contain sufficient information on the instrument run.");
238  }
239  if (!empty() && run_info().flowcell().naming_method() == constants::UnknownTileNamingMethod)
240  INTEROP_THROW(model::invalid_tile_naming_method, "Unknown tile naming method - update your RunInfo.xml");
241  extraction_metric_set_t& extraction_metrics = get_set<extraction_metric>();
242  // Trim excess channel data for imaging table
243  for(extraction_metric_set_t::iterator it = extraction_metrics.begin(); it != extraction_metrics.end();++it)
244  it->trim(run_info().channels().size());
245  }
