35 void cache_error_by_lane_read(I beg,
36  I end,
37  const size_t max_cycle,
38  const std::vector<read_cycle>& cycle_to_read,
39  summary_by_lane_read<float>& summary_by_lane_read)
40  throw( model::index_out_of_bounds_exception )
41 {
42  typedef std::vector<size_t> cycle_vector_t;
43  typedef std::pair<size_t, size_t> key_t;
44  typedef std::pair<float, float> value_t;
45  typedef std::map< key_t, value_t> error_tile_t;
46  typedef std::vector< error_tile_t > error_by_read_tile_t;
47  error_by_read_tile_t tmp (summary_by_lane_read.size());
48  cycle_vector_t max_error_cycle(summary_by_lane_read.size(), 0);
49  for(;beg != end;++beg)
50  {
51  INTEROP_ASSERT(beg->cycle() > 0);
52  INTEROP_ASSERT((beg->cycle()-1) < cycle_to_read.size());
53  if((beg->cycle()-1) >=cycle_to_read.size())
54  throw model::index_out_of_bounds_exception("Cycle exceeds total cycles from Reads in the RunInfo.xml");
55  const read_cycle& read = cycle_to_read[beg->cycle()-1];
56  if(read.cycle_within_read > max_cycle || read.is_last_cycle_in_read) continue;
57  key_t key = std::make_pair(beg->lane(), beg->tile());
58  const size_t read_number = read.number-1;
59  max_error_cycle[read_number] = std::max(max_error_cycle[read_number], static_cast<size_t>(read.cycle_within_read));
60  INTEROP_ASSERTMSG(read_number < tmp.size(), read.number << " " << read.cycle_within_read << ", " << beg->cycle());
61  if(tmp[read_number].find(key) == tmp[read.number-1].end())
62  tmp[read_number][key] = std::make_pair(0.0f, 0.0f);
63  tmp[read_number][key].first+=beg->errorRate();
64  tmp[read_number][key].second+=1;
65  }
66  for(size_t read=0;read<tmp.size();++read)
67  {
68  for(typename error_tile_t::const_iterator ebeg = tmp[read].begin(), eend = tmp[read].end();ebeg != eend;++ebeg)
69  {
70  INTEROP_ASSERT(read < summary_by_lane_read.read_count());
71  const size_t lane = ebeg->first.first-1;
72  if(lane >= summary_by_lane_read.lane_count())
73  INTEROP_THROW(model::index_out_of_bounds_exception, "Lane exceeds number of lanes in RunInfo.xml");
74  if(max_cycle < std::numeric_limits<size_t>::max() && max_error_cycle[read] < max_cycle)
75  summary_by_lane_read(read, lane).push_back(0);
76  else summary_by_lane_read(read, lane).push_back(divide(ebeg->second.first,ebeg->second.second));
77  }
78  }
